54void writeInterpolableParts(std::ostream &OS,
55 const std::vector<InterpolablePart> &Fragments,
56 bool ForceString, const Node &N) {
57 writeBytecode(OS, mkHeader(EK_ConcatStrings, N));
58 writeBytecode(OS, ForceString);
59
60 // Scan the fragments to see how many "Strings" we need to write.
61 // Because interpolations might be incomplete, we'd like to omit them.
62 // "foo${ }"
63 // ^ <--- skip this empty fragments at all.
64 std::size_t Count = 0;
65 for (const InterpolablePart &Frag : Fragments) {
66 switch (Frag.kind()) {
67 case InterpolablePart::SPK_Escaped:
68 Count++;
69 break;
70 case InterpolablePart::SPK_Interpolation:
71 if (Frag.interpolation().expr())
72 Count++;
73 break;
74 }
75 }
76
77 writeBytecode(OS, Count);
78
79 for (const InterpolablePart &Frag : Fragments) {
80 switch (Frag.kind()) {
81 case InterpolablePart::SPK_Escaped:
82 // FIXME: the location is not correct, not the fragment range.
83 writeBytecode(OS, mkHeader(EK_String, N));
84 writeBytecode(OS, Frag.escaped());
85 break;
86 case InterpolablePart::SPK_Interpolation: {
87 const Node *Expr = Frag.interpolation().expr();
88 if (!Expr)
89 continue;
90 writeBytecode(OS, Expr);
91 break;
92 }
93 }
94 }
95}

185void writeExprAttrs(std::ostream &OS, const ExprAttrs &Attrs) {
186 writeBytecode(OS, mkHeader(EK_Attrs, Attrs));
187 writeBytecode(OS, Attrs.isRecursive());
188
189 const SemaAttrs &SA = Attrs.sema();
190
191 std::size_t StaticCount = 0;
192 for (const auto &[Key, Attribute] : SA.staticAttrs()) {
193 if (Attribute.value())
194 StaticCount++;
195 }
196 writeBytecode(OS, StaticCount);
197
198 for (const auto &[Key, Attribute] : SA.staticAttrs()) {
199 if (!Attribute.value())
200 continue;
201 writeBytecode(OS, Key);
202
203 // AttrDef
204 writeBytecode(OS, Attribute.value());
205 writePos(OS, Attribute.key().lCur().position());
206 writeBytecode(OS, Attribute.fromInherit());
207 }
208
209 std::size_t DynamicCount = 0;
210 for (const auto &Attribute : SA.dynamicAttrs()) {
211 if (Attribute.value())
212 DynamicCount++;
213 }
214
215 writeBytecode(OS, DynamicCount);
216
217 for (const auto &Attribute : SA.dynamicAttrs()) {
218 if (!Attribute.value())
219 continue;
220 assert(Attribute.key().kind() == Node::NK_AttrName);
221 const auto &Name = static_cast<const AttrName &>(Attribute.key());
222 switch (Name.kind()) {
223 case AttrName::ANK_ID:
224 assert(false && "dynamic attrname should not be ID!");
225 break;
226 case AttrName::ANK_String:
227 writeExprString(OS, Name.string());
228 break;
229 case AttrName::ANK_Interpolation:
230 writeBytecode(OS, Name.interpolation().expr());
231 break;
232 }
233
234 writeNull(OS, Attribute.value());
235 writePos(OS, Attribute.key().lCur().position());
236 }

248void writeLambdaArg(std::ostream &OS, const LambdaArg *Arg) {
249 if (!Arg) {
250 writeBytecode(OS, false); // HasArg
251 writeBytecode(OS, false); // HasFormals
252 return;
253 }
254
255 writeBytecode(OS, bool(Arg->id())); // HasArg
256 if (Arg->id())
257 writeBytecode(OS, Arg->id()->name());
258
259 writeBytecode(OS, bool(Arg->formals())); // HasFormals
260 if (Arg->formals()) {
261 bool HasEllipsis = false;
262 for (const auto &Formal : Arg->formals()->members()) {
263 if (Formal->isEllipsis())
264 HasEllipsis = true;
265 }
266 writeBytecode(OS, HasEllipsis);
267 writeBytecode(OS, std::size_t(Arg->formals()->dedup().size()));
268 for (const auto &[Name, Formal] : Arg->formals()->dedup()) {
269 writePos(OS, Formal->lCur().position());
270 writeBytecode(OS, Name);
271 writeBytecode(OS, Formal->defaultExpr());
272 }
273 }

346void writeExprBinOp(std::ostream &OS, const ExprBinOp &N) {
347 using namespace tok;
348
349 if (!N.lhs() || !N.rhs()) // Skip invalid nodes
350 return;
351
352 const Expr *LHS = N.lhs();
353 const Expr *RHS = N.rhs();
354
355 switch (N.op().op()) {
356 case tok_op_eq:
357 writeBytecode(OS, mkHeader(EK_OpEq, N));
358 writeNull(OS, LHS);
359 writeNull(OS, RHS);
360 break;
361 case tok_op_neq:
362 writeBytecode(OS, mkHeader(EK_OpNEq, N));
363 writeNull(OS, LHS);
364 writeNull(OS, RHS);
365 break;
366 case tok_op_and:
367 writeBytecode(OS, mkHeader(EK_OpAnd, N));
368 writeNull(OS, LHS);
369 writeNull(OS, RHS);
370 break;
371 case tok_op_or:
372 writeBytecode(OS, mkHeader(EK_OpOr, N));
373 writeNull(OS, LHS);
374 writeNull(OS, RHS);
375 break;
376 case tok_op_impl:
377 writeBytecode(OS, mkHeader(EK_OpImpl, N));
378 writeNull(OS, LHS);
379 writeNull(OS, RHS);
380 break;
381 case tok_op_update:
382 writeBytecode(OS, mkHeader(EK_OpUpdate, N));
383 writeNull(OS, LHS);
384 writeNull(OS, RHS);
385 break;
386 case tok_op_concat:
387 writeBytecode(OS, mkHeader(EK_OpConcatLists, N));
388 writeNull(OS, LHS);
389 writeNull(OS, RHS);
390 break;
391 case tok_op_negate:
392 writeBinOpCall(OS, N, "__sub");
393 writeNull(OS, LHS);
394 writeNull(OS, RHS);
395 break;
396 case tok_op_div:
397 writeBinOpCall(OS, N, "__div");
398 writeNull(OS, LHS);
399 writeNull(OS, RHS);
400 break;
401 case tok_op_mul:
402 writeBinOpCall(OS, N, "__mul");
403 writeNull(OS, LHS);
404 writeNull(OS, RHS);
405 break;
406 case tok_op_lt:
407 // (lt a b)
408 writeBinOpCall(OS, N, "__lessThan");
409 writeNull(OS, LHS);
410 writeNull(OS, RHS);
411 break;
412 case tok_op_le:
413 // (le a b) -> (not (lt b a))
414 writeBytecode(OS, mkHeader(EK_OpNot, N));
415 writeBinOpCall(OS, N, "__lessThan");
416 writeNull(OS, RHS);
417 writeNull(OS, LHS);
418 break;
419 case tok_op_gt:
420 // (gt a b) -> (lt b a)
421 writeBinOpCall(OS, N, "__lessThan");
422 writeNull(OS, RHS);
423 writeNull(OS, LHS);
424 break;
425 case tok_op_ge:
426 // (ge a b) -> (not (lt a b))
427 writeBytecode(OS, mkHeader(EK_OpNot, N));
428 writeBinOpCall(OS, N, "__lessThan");
429 writeNull(OS, LHS);
430 writeNull(OS, RHS);
431 break;
432 case tok_op_add:
433 writeBytecode(OS, mkHeader(nixbc::EK_ConcatStrings, N));
434 writeBytecode(OS, false); // forceString
435 writeBytecode(OS, std::size_t(2));
436 writeNull(OS, LHS);
437 writeNull(OS, RHS);
438 break;
439 default:
440 assert(false && "Unhandled binary operator");
441 __builtin_unreachable();
442 }
