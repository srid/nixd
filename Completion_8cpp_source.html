23/// Set max completion size to this value, we don't want to send large lists
24/// because of slow IO.
25/// Items exceed this size should be marked "incomplete" and recomputed.
26constexpr int MaxCompletionSize = 30;
27
28CompletionItemKind OptionKind = CompletionItemKind::Constructor;
29CompletionItemKind OptionAttrKind = CompletionItemKind::Class;

31struct ExceedSizeError : std::exception {
32 [[nodiscard]] const char *what() const noexcept override {
33 return "Size exceeded";
34 }
35};

37void addItem(std::vector<CompletionItem> &Items, CompletionItem Item) {
38 if (Items.size() >= MaxCompletionSize) {
39 throw ExceedSizeError();
40 }
41 Items.emplace_back(std::move(Item));
42}

44class VLACompletionProvider {
45 const VariableLookupAnalysis &VLA;
46
47 static CompletionItemKind getCompletionItemKind(const Definition &Def) {
48 if (Def.isBuiltin()) {
49 return CompletionItemKind::Keyword;
50 }
51 return CompletionItemKind::Variable;
52 }
53
54 /// Collect definition on some env, and also it's ancestors.
55 void collectDef(std::vector<CompletionItem> &Items, const EnvNode *Env,
56 const std::string &Prefix) {
57 if (!Env)
58 return;
59 collectDef(Items, Env->parent(), Prefix);
60 for (const auto &[Name, Def] : Env->defs()) {
61 if (Name.starts_with(
62 "__")) // These names are nix internal implementation, skip.
63 continue;
64 assert(Def);
65 if (Name.starts_with(Prefix)) {
66 addItem(Items, CompletionItem{
67 .label = Name,
68 .kind = getCompletionItemKind(*Def),
69 });
70 }
71 }
72 }
73
74public:
75 VLACompletionProvider(const VariableLookupAnalysis &VLA) : VLA(VLA) {}
76
77 /// Perform code completion right after this node.
78 void complete(const nixf::Node &Desc, std::vector<CompletionItem> &Items,
79 const ParentMapAnalysis &PM) {
80 std::string Prefix; // empty string, accept all prefixes
81 if (Desc.kind() == Node::NK_Identifer)
82 Prefix = static_cast<const Identifier &>(Desc).name();
83 collectDef(Items, upEnv(Desc, VLA, PM), Prefix);
84 }
85};

87/// \brief Provide completions by IPC. Asking nixpkgs provider.
88/// We simply select nixpkgs in separate process, thus this value does not need
89/// to be cached. (It is already cached in separate process.)
90///
91/// Currently, this procedure is explicitly blocked (synchronized). Because
92/// query nixpkgs value is relatively fast. In the future there might be nixd
93/// index, for performance.
94class NixpkgsCompletionProvider {
95
96 AttrSetClient &NixpkgsClient;
97
98public:
99 NixpkgsCompletionProvider(AttrSetClient &NixpkgsClient)
100 : NixpkgsClient(NixpkgsClient) {}
101
102 void resolvePackage(std::vector<std::string> Scope, std::string Name,
103 CompletionItem &Item) {
104 std::binary_semaphore Ready(0);
105 PackageDescription Desc;
106 auto OnReply = [&Ready, &Desc](llvm::Expected<AttrPathInfoResponse> Resp) {
107 if (Resp)
108 Desc = *Resp;
109 Ready.release();
110 };
111 Scope.emplace_back(std::move(Name));
112 NixpkgsClient.attrpathInfo(Scope, std::move(OnReply));
113 Ready.acquire();
114 // Format "detail" and document.
115 Item.documentation = MarkupContent{
116 .kind = MarkupKind::Markdown,
117 .value = Desc.Description.value_or("") + "\n\n" +
118 Desc.LongDescription.value_or(""),
119 };
120 Item.detail = Desc.Version.value_or("?");
121 }
122
123 /// \brief Ask nixpkgs provider, give us a list of names. (thunks)
124 void completePackages(std::vector<std::string> Scope, std::string Prefix,
125 std::vector<CompletionItem> &Items) {
126 std::binary_semaphore Ready(0);
127 std::vector<std::string> Names;
128 auto OnReply = [&Ready,
129 &Names](llvm::Expected<AttrPathCompleteResponse> Resp) {
130 if (!Resp) {
131 lspserver::elog("nixpkgs evaluator reported: {0}", Resp.takeError());
132 Ready.release();
133 return;
134 }
135 Names = *Resp; // Copy response to waiting thread.
136 Ready.release();
137 };
138 // Send request.
139 AttrPathCompleteParams Params{std::move(Scope), std::move(Prefix)};
140 NixpkgsClient.attrpathComplete(Params, std::move(OnReply));
141 Ready.acquire();
142 // Now we have "Names", use these to fill "Items".
143 for (const auto &Name : Names) {
144 if (Name.starts_with(Prefix)) {
145 addItem(Items, CompletionItem{
146 .label = Name,
147 .kind = CompletionItemKind::Field,
148 .data = llvm::formatv("{0}", toJSON(Params)),
149 });
150 }
151 }
152 }
153};

155/// \brief Provide completion list by nixpkgs module system (options).
156class OptionCompletionProvider {
157 AttrSetClient &OptionClient;
158
159 // Where is the module set. (e.g. nixos)
160 std::string ModuleOrigin;
161
162 // Wheter the client support code snippets.
163 bool ClientSupportSnippet;
164
165 static std::string escapeCharacters(const std::set<char> &Charset,
166 const std::string &Origin) {
167 // Escape characters listed in charset.
168 std::string Ret;
169 Ret.reserve(Origin.size());
170 for (const auto Ch : Origin) {
171 if (Charset.contains(Ch)) {
172 Ret += "\\";
173 Ret += Ch;
174 } else {
175 Ret += Ch;
176 }
177 }
178 return Ret;
179 }
180
181 void fillInsertText(CompletionItem &Item, const std::string &Name,
182 const OptionDescription &Desc) const {
183 if (!ClientSupportSnippet) {
184 Item.insertTextFormat = InsertTextFormat::PlainText;
185 Item.insertText = Name + " = " + Desc.Example.value_or("") + ";";
186 return;
187 }
188 Item.insertTextFormat = InsertTextFormat::Snippet;
189 Item.insertText =
190 Name + " = " +
191 "${1:" + escapeCharacters({'\\', '$', '}'}, Desc.Example.value_or("")) +
192 "}" + ";";
193 }
194
195public:
196 OptionCompletionProvider(AttrSetClient &OptionClient,
197 std::string ModuleOrigin, bool ClientSupportSnippet)
198 : OptionClient(OptionClient), ModuleOrigin(std::move(ModuleOrigin)),
199 ClientSupportSnippet(ClientSupportSnippet) {}
200
201 void completeOptions(std::vector<std::string> Scope, std::string Prefix,
202 std::vector<CompletionItem> &Items) {
203 std::binary_semaphore Ready(0);
204 OptionCompleteResponse Names;
205 auto OnReply = [&Ready,
206 &Names](llvm::Expected<OptionCompleteResponse> Resp) {
207 if (!Resp) {
208 lspserver::elog("option worker reported: {0}", Resp.takeError());
209 Ready.release();
210 return;
211 }
212 Names = *Resp; // Copy response to waiting thread.
213 Ready.release();
214 };
215 // Send request.
216 AttrPathCompleteParams Params{std::move(Scope), std::move(Prefix)};
217 OptionClient.optionComplete(Params, std::move(OnReply));
218 Ready.acquire();
219 // Now we have "Names", use these to fill "Items".
220 for (const nixd::OptionField &Field : Names) {
221 CompletionItem Item;
222
223 Item.label = Field.Name;
224 Item.detail = ModuleOrigin;
225
226 if (Field.Description) {
227 const OptionDescription &Desc = *Field.Description;
228 Item.kind = OptionKind;
229 fillInsertText(Item, Field.Name, Desc);
230 Item.documentation = MarkupContent{
231 .kind = MarkupKind::Markdown,
232 .value = Desc.Description.value_or(""),
233 };
234 Item.detail += " | "; // separater between origin and type desc.
235 if (Desc.Type) {
236 std::string TypeName = Desc.Type->Name.value_or("");
237 std::string TypeDesc = Desc.Type->Description.value_or("");
238 Item.detail += llvm::formatv("{0} ({1})", TypeName, TypeDesc);
239 } else {
240 Item.detail += "? (missing type)";
241 }
242 addItem(Items, std::move(Item));
243 } else {
244 Item.kind = OptionAttrKind;
245 addItem(Items, std::move(Item));
246 }
247 }
248 }
249};

251void completeAttrName(const std::vector<std::string> &Scope,
252 const std::string &Prefix,
253 Controller::OptionMapTy &Options, bool CompletionSnippets,
254 std::vector<CompletionItem> &List) {
255 for (const auto &[Name, Provider] : Options) {
256 AttrSetClient *Client = Options.at(Name)->client();
257 if (!Client) [[unlikely]] {
258 elog("skipped client {0} as it is dead", Name);
259 continue;
260 }
261 OptionCompletionProvider OCP(*Client, Name, CompletionSnippets);
262 OCP.completeOptions(Scope, Prefix, List);
263 }
264}

268void Controller::onCompletion(const CompletionParams &Params,
269 Callback<CompletionList> Reply) {
270 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
271 Pos = toNixfPosition(Params.position), this]() mutable {
272 std::string File(URI.file());
273 if (std::shared_ptr<NixTU> TU = getTU(File, Reply)) [[likely]] {
274 if (std::shared_ptr<nixf::Node> AST = getAST(*TU, Reply)) [[likely]] {
275 const nixf::Node *Desc = AST->descend({Pos, Pos});
276 if (!Desc) {
277 Reply(error("cannot find corresponding node on given position"));
278 return;
279 }
280 CompletionList List;
281 try {
282 const ParentMapAnalysis &PM = *TU->parentMap();
283 std::vector<std::string> Scope;
284 using PathResult = FindAttrPathResult;
285 auto R = findAttrPath(*Desc, PM, Scope);
286 if (R == PathResult::OK) {
287 // Construct request.
288 std::string Prefix = Scope.back();
289 Scope.pop_back();
290 {
291 std::lock_guard _(OptionsLock);
292 completeAttrName(Scope, Prefix, Options,
293 ClientCaps.CompletionSnippets, List.items);
294 }
295 } else {
296 const VariableLookupAnalysis &VLA = *TU->variableLookup();
297 VLACompletionProvider VLAP(VLA);
298 VLAP.complete(*Desc, List.items, PM);
299 if (havePackageScope(*Desc, VLA, PM)) {
300 // Append it with nixpkgs completion
301 // FIXME: handle null nixpkgsClient()
302 NixpkgsCompletionProvider NCP(*nixpkgsClient());
303 auto [Scope, Prefix] = getScopeAndPrefix(*Desc, PM);
304 NCP.completePackages(Scope, Prefix, List.items);
305 }
306 }
307 // Next, add nixpkgs provided names.
308 } catch (ExceedSizeError &Err) {
309 List.isIncomplete = true;
310 }
311 Reply(std::move(List));
312 }
313 }
314 };
315 boost::asio::post(Pool, std::move(Action));
316}

318void Controller::onCompletionItemResolve(const CompletionItem &Params,
319 Callback<CompletionItem> Reply) {
320
321 auto Action = [Params, Reply = std::move(Reply), this]() mutable {
322 if (Params.data.empty()) {
323 Reply(Params);
324 return;
325 }
326 AttrPathCompleteParams Req;
327 auto EV = llvm::json::parse(Params.data);
328 if (!EV) {
329 // If the json value cannot be parsed, this is very unlikely to happen.
330 Reply(EV.takeError());
331 return;
332 }
333
334 llvm::json::Path::Root Root;
335 fromJSON(*EV, Req, Root);
336
337 // FIXME: handle null nixpkgsClient()
338 NixpkgsCompletionProvider NCP(*nixpkgsClient());
339 CompletionItem Resp = Params;
340 NCP.resolvePackage(Req.Scope, Params.label, Resp);
341
342 Reply(std::move(Resp));
343 };
344 boost::asio::post(Pool, std::move(Action));
