38void fillUnsafeGetAttrPosLocation(nix::EvalState &State, nix::Value &V,
39 lspserver::Location &Loc) {
40 State.forceValue(V, nix::noPos);
41 nix::Value &File = nixt::selectAttr(State, V, State.symbols.create("file"));
42 nix::Value &Line = nixt::selectAttr(State, V, State.symbols.create("line"));
43 nix::Value &Column =
44 nixt::selectAttr(State, V, State.symbols.create("column"));
45
46 if (File.type() == nix::ValueType::nString)
47 Loc.uri = URIForFile::canonicalize(File.c_str(), File.c_str());
48
49 if (Line.type() == nix::ValueType::nInt &&
50 Column.type() == nix::ValueType::nInt) {
51
52 // Nix position starts from "1" however lsp starts from zero.
53 lspserver::Position Pos = {static_cast<int>(Line.integer) - 1,
54 static_cast<int>(Column.integer) - 1};
55 Loc.range = {Pos, Pos};
56 }

59void fillOptionDeclarationPositions(nix::EvalState &State, nix::Value &V,
60 OptionDescription &R) {
61 State.forceValue(V, nix::noPos);
62 if (V.type() != nix::ValueType::nList)
63 return;
64 for (nix::Value *Item : V.listItems()) {
65 // Each item should have "column", "line", "file" fields.
66 lspserver::Location Loc;
67 fillUnsafeGetAttrPosLocation(State, *Item, Loc);
68 R.Declarations.emplace_back(std::move(Loc));
69 }

183void AttrSetProvider::onAttrPathComplete(
184 const AttrPathCompleteParams &Params,
185 lspserver::Callback<AttrPathCompleteResponse> Reply) {
186 try {
187 nix::Value &Scope = nixt::selectStrings(state(), Nixpkgs, Params.Scope);
188
189 state().forceValue(Scope, nix::noPos);
190
191 if (Scope.type() != nix::ValueType::nAttrs) {
192 Reply(error("scope is not an attrset"));
193 return;
194 }
195
196 std::vector<std::string> Names;
197 int Num = 0;
198
199 // FIXME: we may want to use "Trie" to speedup the string searching.
200 // However as my (roughtly) profiling the critical in this loop is
201 // evaluating package details.
202 // "Trie"s may not beneficial becausae it cannot speedup eval.
203 for (const auto *AttrPtr :
204 Scope.attrs->lexicographicOrder(state().symbols)) {
205 const nix::Attr &Attr = *AttrPtr;
206 const std::string Name = state().symbols[Attr.name];
207 if (Name.starts_with(Params.Prefix)) {
208 ++Num;
209 Names.emplace_back(Name);
210 // We set this a very limited number as to speedup
211 if (Num > MaxItems)
212 break;
213 }
214 }
215 Reply(std::move(Names));
216 return;
217 } catch (const nix::BaseError &Err) {
218 Reply(error(Err.info().msg.str()));
219 return;

253void AttrSetProvider::onOptionComplete(
254 const AttrPathCompleteParams &Params,
255 lspserver::Callback<OptionCompleteResponse> Reply) {
256 try {
257 nix::Value Scope = nixt::selectOptions(
258 state(), Nixpkgs, nixt::toSymbols(state().symbols, Params.Scope));
259
260 state().forceValue(Scope, nix::noPos);
261
262 if (Scope.type() != nix::ValueType::nAttrs) {
263 Reply(error("scope is not an attrset"));
264 return;
265 }
266
267 if (nixt::isOption(state(), Scope)) {
268 Reply(error("scope is already an option"));
269 return;
270 }
271
272 std::vector<OptionField> Response;
273
274 // FIXME: we may want to use "Trie" to speedup the string searching.
275 // However as my (roughtly) profiling the critical in this loop is
276 // evaluating package details.
277 // "Trie"s may not beneficial becausae it cannot speedup eval.
278 for (const auto *AttrPtr :
279 Scope.attrs->lexicographicOrder(state().symbols)) {
280 const nix::Attr &Attr = *AttrPtr;
281 std::string Name = state().symbols[Attr.name];
282 if (Name.starts_with(Params.Prefix)) {
283 // Add a new "OptionField", see it's type.
284 assert(Attr.value);
285 OptionField NewField;
286 NewField.Name = Name;
287 if (nixt::isOption(state(), *Attr.value)) {
288 OptionDescription Desc;
289 fillOptionDescription(state(), *Attr.value, Desc);
290 NewField.Description = std::move(Desc);
291 }
292 Response.emplace_back(std::move(NewField));
293 // We set this a very limited number as to speedup
294 if (Response.size() >= MaxItems)
295 break;
296 }
297 }
298 Reply(std::move(Response));
299 return;
300 } catch (const nix::BaseError &Err) {
301 Reply(error(Err.info().msg.str()));
302 return;
303 } catch (const std::exception &Err) {
304 Reply(error(Err.what()));
305 return;
