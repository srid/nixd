30lspserver::Range getLambdaSelectionRage(const ExprLambda &Lambda) {
31 if (!Lambda.arg()) {
32 return toLSPRange(Lambda.range());

35 if (!Lambda.arg()->id()) {

42lspserver::Range getAttrRange(const Attribute &Attr) {
43 auto LCur = toLSPPosition(Attr.key().lCur());
44 if (Attr.value())
45 return {LCur, toLSPPosition(Attr.value()->rCur())};
46 return {LCur, toLSPPosition(Attr.key().rCur())};

49/// Make variable's entry rich.
50void richVar(const ExprVar &Var, DocumentSymbol &Sym,
51 const VariableLookupAnalysis &VLA) {
52 if (Var.id().name() == "true" || Var.id().name() == "false") {
53 Sym.kind = SymbolKind::Boolean;
54 Sym.detail = "builtin boolean";
55 return;
56 }
57
58 if (Var.id().name() == "null") {
59 Sym.kind = SymbolKind::Null;
60 Sym.detail = "null";
61 return;
62 }
63
64 auto Result = VLA.query(Var);
65 using ResultKind = VariableLookupAnalysis::LookupResultKind;
66 if (Result.Kind == ResultKind::Defined)
67 Sym.kind = SymbolKind::Constant;
68 else if (Result.Kind == ResultKind::FromWith)
69 Sym.kind = SymbolKind::Variable;
70 else {
71 Sym.deprecated = true;
72 return;
73 }
74
75 if (Result.Def->isBuiltin())
76 Sym.kind = SymbolKind::Event;
77}

79/// Collect document symbol on AST.
80void collect(const Node *AST, std::vector<DocumentSymbol> &Symbols,
81 const VariableLookupAnalysis &VLA) {
82 if (!AST)
83 return;
84 switch (AST->kind()) {
85
86 case Node::NK_ExprString: {
87 const auto &Str = static_cast<const ExprString &>(*AST);
88 DocumentSymbol Sym{
89 .name = Str.isLiteral() ? Str.literal() : "(dynamic string)",
90 .detail = "string",
91 .kind = SymbolKind::String,
92 .deprecated = false,
93 .range = toLSPRange(Str.range()),
94 .selectionRange = toLSPRange(Str.range()),
95 .children = {},
96 };
97 Symbols.emplace_back(std::move(Sym));
98 break;
99 }
100 case Node::NK_ExprInt: {
101 const auto &Int = static_cast<const ExprInt &>(*AST);
102 DocumentSymbol Sym{
103 .name = std::to_string(Int.value()),
104 .detail = "integer",
105 .kind = SymbolKind::Number,
106 .deprecated = false,
107 .range = toLSPRange(Int.range()),
108 .selectionRange = toLSPRange(Int.range()),
109 .children = {},
110 };
111 Symbols.emplace_back(std::move(Sym));
112 break;
113 }
114 case Node::NK_ExprFloat: {
115 const auto &Float = static_cast<const ExprFloat &>(*AST);
116 DocumentSymbol Sym{
117 .name = std::to_string(Float.value()),
118 .detail = "float",
119 .kind = SymbolKind::Number,
120 .deprecated = false,
121 .range = toLSPRange(Float.range()),
122 .selectionRange = toLSPRange(Float.range()),
123 .children = {},
124 };
125 Symbols.emplace_back(std::move(Sym));
126 break;
127 }
128 case Node::NK_AttrName: {
129 const auto &AN = static_cast<const AttrName &>(*AST);
130 DocumentSymbol Sym{
131 .name = AN.isStatic() ? AN.staticName() : "(dynamic attribute name)",
132 .detail = "attribute name",
133 .kind = SymbolKind::Property,
134 .deprecated = false,
135 .range = toLSPRange(AN.range()),
136 .selectionRange = toLSPRange(AN.range()),
137 .children = {},
138 };
139 Symbols.emplace_back(std::move(Sym));
140 break;
141 }
142 case Node::NK_ExprVar: {
143 const auto &Var = static_cast<const ExprVar &>(*AST);
144 DocumentSymbol Sym{
145 .name = Var.id().name(),
146 .detail = "identifier",
147 .kind = SymbolKind::Variable,
148 .deprecated = false,
149 .range = toLSPRange(Var.range()),
150 .selectionRange = toLSPRange(Var.range()),
151 .children = {},
152 };
153 richVar(Var, Sym, VLA);
154 Symbols.emplace_back(std::move(Sym));
155 break;
156 }
157 case Node::NK_ExprLambda: {
158 std::vector<DocumentSymbol> Children;
159 const auto &Lambda = static_cast<const ExprLambda &>(*AST);
160 collect(Lambda.body(), Children, VLA);
161 DocumentSymbol Sym{
162 .name = getLambdaName(Lambda),
163 .detail = "lambda",
164 .kind = SymbolKind::Function,
165 .deprecated = false,
166 .range = toLSPRange(Lambda.range()),
167 .selectionRange = getLambdaSelectionRage(Lambda),
168 .children = std::move(Children),
169 };
170 Symbols.emplace_back(std::move(Sym));
171 break;
172 }
173 case Node::NK_ExprList: {
174 std::vector<DocumentSymbol> Children;
175 const auto &List = static_cast<const ExprList &>(*AST);
176 for (const Node *Ch : AST->children())
177 collect(Ch, Children, VLA);
178
179 DocumentSymbol Sym{
180 .name = "{anonymous}",
181 .detail = "list",
182 .kind = SymbolKind::Array,
183 .deprecated = false,
184 .range = toLSPRange(List.range()),
185 .selectionRange = toLSPRange(List.range()),
186 .children = std::move(Children),
187 };
188 Symbols.emplace_back(std::move(Sym));
189 break;
190 }
191 case Node::NK_ExprAttrs: {
192 const SemaAttrs &SA = static_cast<const ExprAttrs &>(*AST).sema();
193 for (const auto &[Name, Attr] : SA.staticAttrs()) {
194 if (!Attr.value())
195 continue;
196 std::vector<DocumentSymbol> Children;
197 collect(Attr.value(), Children, VLA);
198 DocumentSymbol Sym{
199 .name = Name,
200 .detail = "attribute",
201 .kind = SymbolKind::Field,
202 .deprecated = false,
203 .range = getAttrRange(Attr),
204 .selectionRange = toLSPRange(Attr.key().range()),
205 .children = std::move(Children),
206 };
207 Symbols.emplace_back(std::move(Sym));
208 }
209 for (const nixf::Attribute &Attr : SA.dynamicAttrs()) {
210 std::vector<DocumentSymbol> Children;
211 collect(Attr.value(), Children, VLA);
212 DocumentSymbol Sym{
213 .name = "${dynamic attribute}",
214 .detail = "attribute",
215 .kind = SymbolKind::Field,
216 .deprecated = false,
217 .range = getAttrRange(Attr),
218 .selectionRange = toLSPRange(Attr.key().range()),
219 .children = std::move(Children),
220 };
221 Symbols.emplace_back(std::move(Sym));
222 }
223 break;
224 }
225 default:
226 // Trivial dispatch. Treat these symbol as same as this level.
227 for (const Node *Ch : AST->children())
228 collect(Ch, Symbols, VLA);
229 break;
230 }
231}

235void Controller::onDocumentSymbol(const DocumentSymbolParams &Params,
236 Callback<std::vector<DocumentSymbol>> Reply) {
237 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
238 this]() mutable {
239 if (std::shared_ptr<NixTU> TU =
240 getTU(URI.file().str(), Reply, /*Ignore=*/true)) {
241 if (std::shared_ptr<Node> AST = getAST(*TU, Reply)) {
242 std::vector<DocumentSymbol> Symbols;
243 collect(AST.get(), Symbols, *TU->variableLookup());
244 Reply(std::move(Symbols));
245 }
246 }
247 };
248 boost::asio::post(Pool, std::move(Action));
249}
