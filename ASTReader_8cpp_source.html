19nix::Expr *ASTDeserializer::eatHookable(std::string_view &Data) {
20#pragma GCC diagnostic push
21#pragma GCC diagnostic error "-Wimplicit-fallthrough"
22
23#define HOOKEXPR(TYPE, EXPR) HookExpr##TYPE(EXPR, VMap, EMap, Hdr.Handle)
24 auto Hdr = eat<NodeHeader>(Data);
25 nix::PosIdx Pos = Ctx.PTable.add(Ctx.Origin, Hdr.BeginLine, Hdr.BeginColumn);
26 nix::Expr *NewExpr = nullptr;
27 switch (Hdr.Kind) {
28 case EK_Null:
29 return nullptr;
30 case EK_Int:
31 NewExpr = new HOOKEXPR(Int, eatExprInt(Data));
32 break;
33 case EK_Float:
34 NewExpr = new HOOKEXPR(Float, eatExprFloat(Data));
35 break;
36 case EK_String: {
37 auto S = bc::eat<std::string>(Data);
38 NewExpr = new HOOKEXPR(String, nix::ExprString(std::move(S)));
39 break;
40 }
41 case EK_Path:
42 NewExpr = new HOOKEXPR(Path, eatExprPath(Data));
43 break;
44 case EK_Var: {
45 NewExpr = new HOOKEXPR(Var, eatExprVar(Pos, Data));
46 break;
47 }
48 case EK_Select:
49 NewExpr = new HOOKEXPR(Select, eatExprSelect(Pos, Data));
50 break;
51 case EK_OpHasAttr:
52 NewExpr = new HOOKEXPR(OpHasAttr, eatExprOpHasAttr(Data));
53 break;
54 case EK_Attrs:
55 NewExpr = new HOOKEXPR(Attrs, eatExprAttrs(Pos, Data));
56 break;
57 case EK_List:
58 NewExpr = new HOOKEXPR(List, eatExprList(Data));
59 break;
60 case EK_Lambda:
61 NewExpr = new HOOKEXPR(Lambda, eatExprLambnda(Pos, Data));
62 break;
63 case EK_Call:
64 NewExpr = new HOOKEXPR(Call, eatExprCall(Pos, Data));
65 break;
66 case EK_Let:
67 NewExpr = new HOOKEXPR(Let, eatExprLet(Data));
68 break;
69 case EK_With:
70 NewExpr = new HOOKEXPR(With, eatExprWith(Pos, Data));
71 break;
72 case EK_If:
73 NewExpr = new HOOKEXPR(If, eatExprIf(Pos, Data));
74 break;
75 case EK_Assert:
76 NewExpr = new HOOKEXPR(Assert, eatExprAssert(Pos, Data));
77 break;
78 case EK_ConcatStrings:
79 NewExpr = new HOOKEXPR(ConcatStrings, eatExprConcatStrings(Pos, Data));
80 break;
81 case EK_OpNot:
82 NewExpr = new HOOKEXPR(OpNot, eatExprOpNot(Data));
83 break;
84#define BinOp(OP, SYM) \
85 case EK_##OP: \
86 NewExpr = new HOOKEXPR(OP, eatExpr##OP(Pos, Data)); \
87 break;
88#include "nixt/BinOps.inc"
89#undef BinOp
90 default:
91 break;
92 }
93#pragma GCC diagnostic pop
94
95 assert(NewExpr && "Unknown hookable kind!");
96 Pool.record(NewExpr);
97 return NewExpr;
98}

131nix::AttrPath ASTDeserializer::eatAttrPath(std::string_view &Data) {
132 auto Size = eat<std::size_t>(Data);
133
134 nix::AttrPath Result;
135 for (std::size_t I = 0; I < Size; I++) {
136 // It could be a symbol, or an expression.
137 // The convention is, if it is a symbol, read an "ExprString"
138 // Otherwise interpret it as an expression (needs to be evaluated).
139 nix::Expr *Expr = eatHookable(Data);
140 if (!Expr)
141 continue;
142 if (const auto *String = dynamic_cast<nix::ExprString *>(Expr)) {
143 // Create a symbol
144 Result.emplace_back(Ctx.STable.create(String->s));
145 } else {
146 Result.emplace_back(Expr);
147 }
148 }
149
150 return Result;
151}

169nix::ExprAttrs ASTDeserializer::eatExprAttrs(nix::PosIdx Pos,
170 std::string_view &Data) {
171 nix::ExprAttrs Result(Pos);
172
173 Result.recursive = eat<bool>(Data);
174
175 auto NStaticDefs = eat<std::size_t>(Data);
176 for (std::size_t I = 0; I < NStaticDefs; I++) {
177 // Read static attr defs
178 auto Name = eatSymbol(Data);
179
180 // AttrDef
181 nix::Expr *E = eatHookable(Data);
182 assert(E);
183 nix::PosIdx Pos = eatPos(Data);
184 auto Inherited = eat<bool>(Data);
185 Result.attrs.insert({Name, {E, Pos, Inherited}});
186 }
187
188 auto NDynamicDefs = eat<std::size_t>(Data);
189 for (std::size_t I = 0; I < NDynamicDefs; I++) {
190 nix::Expr *Name = eatHookable(Data);
191 nix::Expr *Value = eatHookable(Data);
192 assert(Name && Value);
193 nix::PosIdx Pos = eatPos(Data);
194 Result.dynamicAttrs.emplace_back(Name, Value, Pos);
195 }
196
197 return Result;
198}

214nix::ExprLambda ASTDeserializer::eatExprLambnda(nix::PosIdx Pos,
215 std::string_view &Data) {
216 // Does this lambda have a unified argument?
217 // a: a + 1
218 // ^<-------- this symbol
219 std::optional<nix::Symbol> Arg;
220 auto HasArg = eat<bool>(Data);
221
222 if (HasArg)
223 Arg = eatSymbol(Data);
224
225 // Does this lambda have formal arguments?
226 // { a, b }: a + b
227 auto HasFormals = eat<bool>(Data);
228 nix::Formals *Formals = nullptr;
229
230 if (HasFormals) {
231 // Decode the formals
232 // FIXME: !! Apparently leaks here.
233 Formals = new nix::Formals;
234 Formals->ellipsis = eat<bool>(Data);
235
236 auto N = eat<std::size_t>(Data);
237 for (std::size_t I = 0; I < N; I++) {
238 nix::PosIdx Pos = eatPos(Data);
239 nix::Symbol Name = eatSymbol(Data);
240 // explicitly nullable, nullptr means there is no default
241 nix::Expr *Def = /*nullable*/ eatHookable(Data);
242 Formals->formals.emplace_back(Pos, Name, Def);
243 }
244 }
245
246 nix::Expr *Body = eatNonNullHookable(Data);
247
248 if (HasArg)
249 return {Pos, *Arg, Formals, Body};
250 return {Pos, Formals, Body};
251}
