42void EvalProvider::onRegisterBC(const rpc::RegisterBCParams &Params) {
43 // Path context
44 auto CachePath = State->rootPath(nix::CanonPath(Params.CachePath));
45
46 nixt::DeserializeContext Ctx = nixt::getDeserializeContext(
47 *State, Params.BasePath, nix::Pos::none_tag{});
48
49 // Extract the buffer from `Params`
50 bipc::shared_memory_object Shm(bipc::open_only, Params.Shm.c_str(),
51 bipc::read_only);
52
53 bipc::mapped_region Region(Shm, bipc::read_only);
54
55 std::string_view RegionView = {(char *)Region.get_address(),
56 (char *)Region.get_address() + Params.Size};
57
58 nix::Expr *AST = nixt::deserializeHookable(RegionView, Ctx, Pool, VMap, EMap);
59
60 nix::Value V;
61 State->eval(AST, V);
62
63 // Inject pre-parsed AST into EvalState cache
64 auto &Cache = nixt::getFileEvalCache(*State);
65 Cache[CachePath] = V;
66}
