20llvm::Expected<WorkspaceEdit> rename(const nixf::Node &Desc,
21 const std::string &NewText,
22 const ParentMapAnalysis &PMA,
23 const VariableLookupAnalysis &VLA,
24 const URIForFile &URI) {
25 using lspserver::TextEdit;
26 // Find "definition"
27 auto Def = findDefinition(Desc, PMA, VLA);
28 if (!Def)
29 return Def.takeError();
30
31 if (Def->source() == Definition::DS_With)
32 return error("cannot rename `with` defined variables");
33
34 if (Def->isBuiltin())
35 return error("cannot rename builtin variable");
36
37 std::vector<TextEdit> Edits;
38
39 for (const auto *Use : Def->uses()) {
40 Edits.emplace_back(TextEdit{
41 .range = toLSPRange(Use->range()),
42 .newText = NewText,
43 });
44 }
45
46 Edits.emplace_back(TextEdit{
47 .range = toLSPRange(Def->syntax()->range()),
48 .newText = NewText,
49 });
50 WorkspaceEdit WE;
51 WE.changes = std::map<std::string, std::vector<TextEdit>>{
52 {URI.uri(), std::move(Edits)}};
53 return WE;
54}

57void Controller::onRename(const RenameParams &Params,

59 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
60 Pos = toNixfPosition(Params.position),
61 NewText = Params.newName, this]() mutable {
62 std::string File(URI.file());
63 if (std::shared_ptr<NixTU> TU = getTU(File, Reply)) [[likely]] {
64 if (std::shared_ptr<nixf::Node> AST = getAST(*TU, Reply)) [[likely]] {
65 const nixf::Node *Desc = AST->descend({Pos, Pos});
66 if (!Desc) {
67 Reply(error("cannot find corresponding node on given position"));
68 return;

70 Reply(rename(*Desc, NewText, *TU->parentMap(), *TU->variableLookup(),
