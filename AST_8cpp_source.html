14/// \brief Find nested attrpath.
15/// e.g. a.b.c.d
16/// ^<- a.b.c.d
17/// { a = 1; b = { c = d; }; }
18/// ^ b.c
19///
20/// SelectAttrPath := { a.b.c.d = 1; }
21/// ^~~~~~~<--------- such "selection"
22/// ValueAttrPath := N is a "value", find how it's nested.
23void getSelectAttrPath(const nixf::AttrName &N,
24 const nixf::ParentMapAnalysis &PM,
25 std::vector<std::string> &Path) {
26 const nixf::Node *Up = PM.query(N);
27 assert(Up && "Naked attrname!");
28 assert(Up->kind() == Node::NK_AttrPath &&
29 "Invoked in non-attrpath name! (Slipped inherit?)");
30 const auto &APath = static_cast<const nixf::AttrPath &>(*Up);
31 // Iterate on attr names
32 Path.reserve(APath.names().size());
33 for (const auto &Name : APath.names()) {
34 if (!Name->isStatic())
35 throw AttrPathHasDynamicError();
36 Path.emplace_back(Name->staticName());
37 if (Name.get() == &N)
38 break;
39 }
40}

42/// \copydoc getSelectAttrPath
43void getValueAttrPath(const nixf::Node &N, const nixf::ParentMapAnalysis &PM,
44 std::vector<std::string> &Path) {
45 if (PM.isRoot(N))
46 return;
47
48 const nixf::Node *Up = PM.query(N);
49 if (!Up)
50 return;
51
52 Up = PM.upExpr(*Up);
53
54 if (!Up)
55 return;
56
57 // Only attrs can "nest something"
58 if (Up->kind() != Node::NK_ExprAttrs)
59 return;
60
61 std::vector<std::string_view> Basic;
62 // Recursively search up all nested entries
63 if (!PM.isRoot(*Up))
64 getValueAttrPath(*Up, PM, Path);
65
66 // Find out how "N" gets nested.
67 const auto &UpAttrs = static_cast<const nixf::ExprAttrs &>(*Up);
68 assert(UpAttrs.binds() && "empty binds cannot nest anything!");
69 for (const std::shared_ptr<nixf::Node> &Attr : UpAttrs.binds()->bindings()) {
70 assert(Attr);
71 if (Attr->kind() == Node::NK_Inherit) // Cannot deal with inherits
72 continue;
73 assert(Attr->kind() == Node::NK_Binding);
74 const auto &Binding = static_cast<const nixf::Binding &>(*Attr);
75 if (Binding.value().get() == &N) {
76 getSelectAttrPath(*Binding.path().names().back(), PM, Path);
77 return;
78 }
79 }
80 assert(false && "must have corresbonding value");
81 __builtin_unreachable();
82}

104bool nixd::havePackageScope(const Node &N, const VariableLookupAnalysis &VLA,
105 const ParentMapAnalysis &PM) {
106 // Firstly find the first "env" enclosed with this variable.
107 const EnvNode *Env = upEnv(N, VLA, PM);
108 if (!Env)
109 return false;
110
111 // Then, search up until there are some `with`.
112 for (; Env; Env = Env->parent()) {
113 if (!Env->isWith())
114 continue;
115 // this env is "with" expression.
116 const Node *With = Env->syntax();
117 assert(With && With->kind() == Node::NK_ExprWith);
118 const Node *WithBody = static_cast<const ExprWith &>(*With).with();
119 if (!WithBody)
120 continue; // skip incomplete with epxression.
121
122 // Se if it is "ExprVar“. Stupid.
123 if (WithBody->kind() != Node::NK_ExprVar)
124 continue;
125
126 // Hardcoded "pkgs", even more stupid.
127 if (static_cast<const ExprVar &>(*WithBody).id().name() == "pkgs")
128 return true;
129 }
130 return false;
131}

143nixd::FindAttrPathResult nixd::findAttrPath(const nixf::Node &N,
144 const nixf::ParentMapAnalysis &PM,
145 std::vector<std::string> &Path) {
146 using R = nixd::FindAttrPathResult;
147 // If this is in "inherit", don't consider it is an attrpath.
148 if (PM.upTo(N, Node::NK_Inherit))
149 return R::Inherit;
150
151 if (const Node *Name = PM.upTo(N, Node::NK_AttrName)) {
152 try {
153 getNestedAttrPath(static_cast<const AttrName &>(*Name), PM, Path);
154 } catch (AttrPathHasDynamicError &E) {
155 return R::WithDynamic;
156 }
157 return R::OK;
158 }
159
160 // Consider this is an "extra" dot.
161 if (const Node *DotNode = PM.upTo(N, Node::NK_Dot)) {
162 const auto &D = static_cast<const Dot &>(*DotNode);
163
164 if (D.prev().kind() != Node::NK_AttrName)
165 return R::NotAttrPath;
166
167 try {
168 getNestedAttrPath(static_cast<const AttrName &>(D.prev()), PM, Path);
169 Path.emplace_back("");
170 return R::OK;
171 } catch (AttrPathHasDynamicError &E) {
172 return R::WithDynamic;
173 }
174 }
175
176 return R::NotAttrPath;
177}
