22Error highlight(const nixf::Node &Desc, const ParentMapAnalysis &PMA,
23 const VariableLookupAnalysis &VLA, const URIForFile &URI,
24 std::vector<DocumentHighlight> &Highlights) {
25 // Find "definition"
26 if (auto Def = findDefinition(Desc, PMA, VLA)) {
27 // OK, iterate all uses.
28 for (const auto *Use : Def->uses()) {
29 assert(Use);
30 Highlights.emplace_back(DocumentHighlight{
31 .range = toLSPRange(Use->range()),
32 .kind = DocumentHighlightKind::Read,
33 });
34 }
35 if (Def->syntax()) {
36 const Node &Syntax = *Def->syntax();
37 Highlights.emplace_back(DocumentHighlight{
38 .range = toLSPRange(Syntax.range()),
39 .kind = DocumentHighlightKind::Write,
40 });
41 }
42 return Error::success();
43 }
44 return error("Cannot find definition of this node");
45}
46} // namespace

48void Controller::onDocumentHighlight(
49 const TextDocumentPositionParams &Params,
50 Callback<std::vector<DocumentHighlight>> Reply) {
51 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
52 Pos = toNixfPosition(Params.position), this]() mutable {
53 std::string File(URI.file());
54 if (std::shared_ptr<NixTU> TU = getTU(File, Reply)) [[likely]] {
55 if (std::shared_ptr<nixf::Node> AST = getAST(*TU, Reply)) [[likely]] {
56 const nixf::Node *Desc = AST->descend({Pos, Pos});
57 if (!Desc) {
58 Reply(error("cannot find corresponding node on given position"));
59 return;
60 }
61 std::vector<DocumentHighlight> Highlights;
62 if (auto Err = highlight(*Desc, *TU->parentMap(), *TU->variableLookup(),
63 URI, Highlights)) {
64 // FIXME: Empty response if there are no def-use chain found.
65 // For document highlights, the specification explicitly specified LSP
66 // should do "fuzzy" things.
67
68 // Empty response on error, don't reply all errors because this method
69 // is very frequently called.
70 Reply(std::vector<DocumentHighlight>{});
71 lspserver::elog("textDocument/documentHighlight failed: {0}", Err);
72 return;
73 }
74 Reply(std::move(Highlights));
75 }
76 }
77 };
78 boost::asio::post(Pool, std::move(Action));
