15void Controller::publishDiagnostics(
16 PathRef File, std::optional<int64_t> Version,
17 const std::vector<nixf::Diagnostic> &Diagnostics) {
18 std::vector<Diagnostic> LSPDiags;
19 LSPDiags.reserve(Diagnostics.size());
20 for (const nixf::Diagnostic &D : Diagnostics) {
21 // Format the message.
22 std::string Message = D.format();
23
24 // Add fix information.
25 if (!D.fixes().empty()) {
26 Message += " (";
27 if (D.fixes().size() == 1) {
28 Message += "fix available";
29 } else {
30 Message += std::to_string(D.fixes().size());
31 Message += " fixes available";
32 }
33 Message += ")";
34 }
35
36 Diagnostic &Diag = LSPDiags.emplace_back(Diagnostic{
37 .range = toLSPRange(D.range()),
38 .severity = getLSPSeverity(D.kind()),
39 .code = D.sname(),
40 .source = "nixf",
41 .message = Message,
42 .tags = toLSPTags(D.tags()),
43 .relatedInformation = std::vector<DiagnosticRelatedInformation>{},
44 });
45
46 assert(Diag.relatedInformation && "Must be initialized");
47 Diag.relatedInformation->reserve(D.notes().size());
48 for (const nixf::Note &N : D.notes()) {
49 Diag.relatedInformation->emplace_back(DiagnosticRelatedInformation{
50 .location =
51 Location{
52 .uri = URIForFile::canonicalize(File, File),
53 .range = toLSPRange(N.range()),
54 },
55 .message = N.format(),
56 });
57 }
58 auto Notes = D.notes();
59 auto DRange = Diag.range;
60
61 for (const nixf::Note &N : Notes) {
62 LSPDiags.emplace_back(Diagnostic{
63 .range = toLSPRange(N.range()),
64 .severity = 4,
65 .code = N.sname(),
66 .source = "nixf",
67 .message = N.format(),
68 .tags = toLSPTags(N.tags()),
69 .relatedInformation =
70 std::vector<DiagnosticRelatedInformation>{
71 DiagnosticRelatedInformation{
72 .location =
73 Location{
74 .uri = URIForFile::canonicalize(File, File),
75 .range = DRange,
76 },
77 .message = "original diagnostic",
78 }},
79 });
80 }
81 }
82 PublishDiagnostic({
83 .uri = URIForFile::canonicalize(File, File),
84 .diagnostics = std::move(LSPDiags),
85 .version = Version,
86 });
87}
