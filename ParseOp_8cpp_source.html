31std::pair<unsigned, unsigned> getBP(TokenKind Kind) {
32 switch (Kind) {
33 case tok_op_impl: // %right ->
34 return {2, 1};
35 case tok_op_or: // %left ||
36 return {3, 4};
37 case tok_op_and: // %left &&
38 return {5, 6};
39 case tok_op_eq: // %nonassoc == !=
40 case tok_op_neq:
41 return {7, 7};
42 case tok_op_lt: // %nonassoc < > <= >=
43 case tok_op_le:
44 case tok_op_ge:
45 case tok_op_gt:
46 return {8, 8};
47 case tok_op_update: // %right //
48 return {10, 9};
49 // %left NOT - 11
50 case tok_op_add: // %left + -
51 case tok_op_negate:
52 return {12, 13};
53 case tok_op_mul: // %left * /
54 return {14, 15};
55 case tok_op_div:
56 case tok_op_concat: // %right ++
57 return {17, 16};
58 // % op_negate
59 default:
60 __builtin_unreachable();
61 }
62}

77std::shared_ptr<Expr> Parser::parseExprOpBP(unsigned LeftRBP) {
78 std::shared_ptr<Expr> Prefix;
79 LexerCursor LCur = lCur();
80 switch (Token Tok = peek(); Tok.kind()) {
81 case tok_op_not:
82 case tok_op_negate: {
83 consume();
84 assert(LastToken && "consume() should have set LastToken");
85 auto O = std::make_shared<Op>(Tok.range(), Tok.kind());
86 auto Expr = parseExprOpBP(getUnaryBP(Tok.kind()));
87 if (!Expr)
88 diagNullExpr(Diags, LastToken->rCur(),
89 "unary operator " + std::string(tok::spelling(Tok.kind())));
90 Prefix =
91 std::make_shared<ExprUnaryOp>(LexerCursorRange{LCur, LastToken->rCur()},
92 std::move(O), std::move(Expr));
93 break;
94 }
95 default:
96 Prefix = parseExprApp();
97 }
98
99 if (!Prefix)
100 return nullptr;
101
102 for (;;) {
103 switch (Token Tok = peek(); Tok.kind()) {
104#define TOK_BIN_OP(NAME) case tok_op_##NAME:
105#include "nixf/Basic/TokenKinds.inc"
106#undef TOK_BIN_OP
107 {
108 // For all binary ops:
109 //
110 // expr_op OP expr_op OP expr_op
111 // ^LeftRBP ^ LBP
112 // |
113 // | we are here
114 auto [LBP, RBP] = getBP(Tok.kind());
115 if (LeftRBP > LBP)
116 return Prefix;
117 if (LeftRBP == LBP) {
118 // TODO: noassoc
119 }
120 consume();
121 assert(LastToken && "consume() should have set LastToken");
122 auto O = std::make_shared<Op>(Tok.range(), Tok.kind());
123 auto RHS = parseExprOpBP(RBP);
124 if (!RHS) {
125 diagNullExpr(Diags, LastToken->rCur(), "binary op RHS");
126 continue;
127 }
128 LexerCursorRange Range{Prefix->lCur(), RHS->rCur()};
129 Prefix = std::make_shared<ExprBinOp>(Range, std::move(O),
130 std::move(Prefix), std::move(RHS));
131 break;
132 }
133 case tok_question: {
134 // expr_op '?' attrpath
135 consume();

137 auto O = std::make_shared<Op>(Tok.range(), Tok.kind());

141 Prefix = std::make_shared<ExprOpHasAttr>(

144 }
