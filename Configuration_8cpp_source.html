49void Controller::updateConfig(Configuration NewConfig) {
50 std::lock_guard G(ConfigLock);
51 Config = std::move(NewConfig);
52
53 if (!Config.nixpkgs.expr.empty()) {
54 /// Evaluate nixpkgs and options, using user-provided config.
55 if (nixpkgsClient()) {
56 evalExprWithProgress(*nixpkgsClient(), Config.nixpkgs.expr,
57 "nixpkgs entries");
58 }
59 }
60 if (!Config.options.empty()) {
61 std::lock_guard _(OptionsLock);
62 // For each option configuration, update the worker.
63 for (const auto &[Name, Opt] : Config.options) {
64 auto &Client = Options[Name];
65 if (!Client) {
66 // If it does not exist. Launch a new client.
67 startOption(Name, Client);
68 }
69 assert(Client);
70 evalExprWithProgress(*Client->client(), Opt.expr, Name);
71 }
72 }
73}

75void Controller::fetchConfig() {
76 auto Action = [this](llvm::Expected<llvm::json::Value> Resp) mutable {
77 if (!Resp) {
78 elog("workspace/configuration: {0}", Resp.takeError());
79 return;
80 }
81
82 // LSP response is a json array, just take the first.
83 if (Resp->kind() != llvm::json::Value::Array) {
84 lspserver::elog("workspace/configuration response is not an array: {0}",
85 *Resp);
86 return;
87 }
88 const Value &FirstConfig = Resp->getAsArray()->front();
89
90 // Run this job in the thread pool. Don't block input thread.
91 auto ConfigAction = [this, FirstConfig]() mutable {
92 // Parse the config
93 Configuration NewConfig;
94 llvm::json::Path::Root P;
95 if (!fromJSON(FirstConfig, NewConfig, P)) {
96 elog("workspace/configuration: parse error {0}", P.getError());
97 return;
98 }
99
100 // OK, update the config
101 updateConfig(std::move(NewConfig));
102 };
103
104 boost::asio::post(Pool, std::move(ConfigAction));
105 };
106 workspaceConfiguration({.items = {ConfigurationItem{.section = "nixd"}}},
107 std::move(Action));
