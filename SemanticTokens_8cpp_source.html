54class SemanticTokenBuilder {

64 void addImpl(nixf::LexerCursor Pos, unsigned Length, unsigned TokenType,
65 unsigned TokenModifiers) {

67 {static_cast<int>(Pos.line()), static_cast<int>(Pos.column())},

73 void add(const Node &N, unsigned TokenType, unsigned TokenModifiers) {
74 if (skip(N))
75 return;
76 addImpl(N.lCur(), len(N), TokenType, TokenModifiers);
77 }

79 static bool skip(const Node &N) {
80 // Skip cross-line strings.
81 return N.range().lCur().line() != N.range().rCur().line();
82 }
83
84 static unsigned len(const Node &N) {
85 return N.range().rCur().offset() - N.range().lCur().offset();
86 }

88 void dfs(const ExprString &Str) {
89 unsigned Modifers = 0;
90 if (!Str.isLiteral())
91 return;
92 add(Str, ST_String, Modifers);
93 }
94
95 void dfs(const ExprVar &Var) {
96 if (Var.id().name() == "true" || Var.id().name() == "false") {
97 add(Var, ST_Bool, SM_Builtin);
98 return;
99 }
100
101 if (Var.id().name() == "null") {
102 add(Var, ST_Null, 0);
103 return;
104 }
105
106 auto Result = VLA.query(Var);
107 using ResultKind = VariableLookupAnalysis::LookupResultKind;
108 if (Result.Def && Result.Def->isBuiltin()) {
109 add(Var, ST_Builtin, SM_Builtin);
110 return;
111 }
112 if (Result.Kind == ResultKind::Defined) {
113 add(Var, ST_Defined, 0);
114 return;
115 }
116 if (Result.Kind == ResultKind::FromWith) {
117 add(Var, ST_FromWith, SM_Dynamic);
118 return;
119 }
120
121 add(Var, ST_Defined, SM_Deprecated);
122 }
123
124 void dfs(const ExprSelect &Select) {
125 dfs(&Select.expr());
126 dfs(Select.defaultExpr());
127 if (!Select.path())
128 return;
129 for (const std::shared_ptr<nixf::AttrName> &Name : Select.path()->names()) {
130 if (!Name)
131 continue;
132 const AttrName &AN = *Name;
133 if (AN.isStatic()) {
134 if (AN.kind() == AttrName::ANK_ID) {
135 add(AN, ST_Select, 0);
136 }
137 }
138 }
139 }
140
141 void dfs(const SemaAttrs &SA) {
142 for (const auto &[Name, Attr] : SA.staticAttrs()) {
143 if (!Attr.value())
144 continue;
145 add(Attr.key(), ST_AttrName, 0);
146 dfs(Attr.value());
147 }
148 for (const auto &Attr : SA.dynamicAttrs()) {
149 dfs(Attr.value());
150 }
151 }
152
153 void dfs(const LambdaArg &Arg) {
154 if (Arg.id())
155 add(*Arg.id(), ST_LambdaArg, 0);
156 // Color deduplicated formals.
157 if (Arg.formals())
158 for (const auto &[_, Formal] : Arg.formals()->dedup()) {
159 if (Formal->id()) {
160 add(*Formal->id(), ST_LambdaFormal, 0);
161 }
162 }
163 }
164
165 void dfs(const ExprLambda &Lambda) {
166 if (Lambda.arg()) {
167 dfs(*Lambda.arg());
168 }
169 dfs(Lambda.body());
170 }
171
172 void dfs(const Node *AST) {
173 if (!AST)
174 return;
175 switch (AST->kind()) {
176 case Node::NK_ExprLambda: {
177 const auto &Lambda = static_cast<const ExprLambda &>(*AST);
178 dfs(Lambda);
179 break;
180 }
181 case Node::NK_ExprString: {
182 const auto &Str = static_cast<const ExprString &>(*AST);
183 dfs(Str);
184 break;
185 }
186 case Node::NK_ExprVar: {
187 const auto &Var = static_cast<const ExprVar &>(*AST);
188 dfs(Var);
189 break;
190 }
191 case Node::NK_ExprSelect: {
192 const auto &Select = static_cast<const ExprSelect &>(*AST);
193 dfs(Select);
194 break;
195 }
196 case Node::NK_ExprAttrs: {
197 const SemaAttrs &SA = static_cast<const ExprAttrs &>(*AST).sema();
198 dfs(SA);
199 break;
200 }
201 default:
202 for (const Node *Ch : AST->children()) {
203 dfs(Ch);
204 }
205 }
206 };

208 std::vector<SemanticToken> finish() {
209 std::vector<SemanticToken> Tokens;
210 std::sort(Raw.begin(), Raw.end());
211 lspserver::Position Prev{0, 0};
212 for (auto Elm : Raw) {
213 assert(Elm.Pos.line - Prev.line >= 0);
214 unsigned DeltaLine = Elm.Pos.line - Prev.line;
215 unsigned DeltaCol =
216 DeltaLine ? Elm.Pos.character : Elm.Pos.character - Prev.character;
217 Prev = Elm.Pos;
218 Tokens.emplace_back(DeltaLine, DeltaCol, Elm.Length, Elm.TokenType,
219 Elm.TokenModifiers);
220 }
221 return Tokens;
222 }
223};

227void Controller::onSemanticTokens(const SemanticTokensParams &Params,
228 Callback<SemanticTokens> Reply) {
229 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
230 this]() mutable {
231 if (std::shared_ptr<NixTU> TU =
232 getTU(URI.file().str(), Reply, /*Ignore=*/true)) {
233 if (std::shared_ptr<Node> AST = getAST(*TU, Reply)) {
234 SemanticTokenBuilder Builder(*TU->variableLookup());
235 Builder.dfs(AST.get());
236 Reply(SemanticTokens{.tokens = Builder.finish()});
237 }
238 }
239 };
240 boost::asio::post(Pool, std::move(Action));
241}
