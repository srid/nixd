20/// Remove the text at `OldRange`, and replace it as `NewText`
21/// Special cases:
22/// 1. Insertions: special `OldRange` that `Begin` == `End`.
23/// 2. Removals: empty `NewText`.
24class TextEdit {
25 LexerCursorRange OldRange;
26 std::string NewText;
27
28public:
29 TextEdit(LexerCursorRange OldRange, std::string NewText)
30 : OldRange(OldRange), NewText(std::move(NewText)) {
31 assert(OldRange.lCur() != OldRange.rCur() || !this->NewText.empty());
32 }
33
34 static TextEdit mkInsertion(LexerCursor P, std::string NewText) {
35 return {{P, P}, std::move(NewText)};
36 }
37
38 static TextEdit mkRemoval(LexerCursorRange RemovingRange) {
39 return {RemovingRange, ""};
40 }
41
42 [[nodiscard]] bool isReplace() const {
43 return !isRemoval() && !isInsertion();
44 }
45
46 [[nodiscard]] bool isRemoval() const { return NewText.empty(); }
47
48 [[nodiscard]] bool isInsertion() const {
49 return OldRange.lCur() == OldRange.rCur();
50 }
51
52 [[nodiscard]] LexerCursorRange oldRange() const { return OldRange; }
53 [[nodiscard]] std::string_view newText() const { return NewText; }
54};

144/// The super class for all diagnostics.
145/// concret diagnostic types are defined in Diagnostic*.inc
146class Diagnostic : public PartialDiagnostic {
147public:
148 /// Each diagnostic contains a severity field,
149 /// should be "Fatal", "Error" or "Warning"
150 /// this will affect the eval process.
151 ///
152 /// "Fatal" -- shouldn't eval the code, e.g. parsing error.
153 /// "Error" -- trigger an error in nix, but we can recover & eval the code.
154 /// "Warning" -- just a warning.
155 /// "Note" -- some additional information about the error.
156 enum Severity { DS_Fatal, DS_Error, DS_Warning };
157
158 /// Internal kind
159 enum DiagnosticKind {
160#define DIAG(SNAME, CNAME, SEVERITY, MESSAGE) DK_##CNAME,
161#include "DiagnosticKinds.inc"
162#undef DIAG
163 };
164
165 Diagnostic(DiagnosticKind Kind, LexerCursorRange Range)
166 : PartialDiagnostic(Range), Kind(Kind) {}
167
168 [[nodiscard]] DiagnosticKind kind() const { return Kind; };
169
170 [[nodiscard]] static Severity severity(DiagnosticKind Kind);
171
172 [[nodiscard]] static const char *message(DiagnosticKind Kind);
173
174 [[nodiscard]] const char *message() const override { return message(kind()); }
175
176 /// Short name, switch name.
177 /// There might be a human readable short name that controls the diagnostic
178 /// For example, one may pass -Wno-dup-formal to suppress duplicated formals.
179 /// A special case for parsing errors, generated from bison

181 [[nodiscard]] static const char *sname(DiagnosticKind Kind);
