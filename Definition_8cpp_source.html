33const Definition *findSelfDefinition(const Node &N,
34 const ParentMapAnalysis &PMA,
35 const VariableLookupAnalysis &VLA) {
36 // If "N" is a definition itself, just return it.
37 if (const Definition *Def = VLA.toDef(N))
38 return Def;
39
40 // If N is inside an attrset, it maybe an "AttrName", let's look for it.
41 const Node *Parent = PMA.query(N);
42 if (Parent && Parent->kind() == Node::NK_AttrName)
43 return VLA.toDef(*Parent);
44
45 return nullptr;
46}

48// Special case, variable in "inherit"
49// inherit name
50// ^~~~<--- this is an "AttrName", not variable.
51const ExprVar *findInheritVar(const Node &N, const ParentMapAnalysis &PMA,
52 const VariableLookupAnalysis &VLA) {
53 if (const Node *Up = PMA.upTo(N, Node::NK_Inherit)) {
54 const Node *UpAn = PMA.upTo(N, Node::NK_AttrName);
55 if (!UpAn)
56 return nullptr;
57 const auto &Inh = static_cast<const Inherit &>(*Up);
58 const auto &AN = static_cast<const AttrName &>(*UpAn);
59
60 // Skip:
61 //
62 // inherit (expr) name1 name2;
63 //
64 if (Inh.expr())
65 return nullptr;
66
67 // Skip dynamic.
68 if (!AN.isStatic())
69 return nullptr;
70
71 // This attrname will be desugared into an "ExprVar".
72 Up = PMA.upTo(Inh, Node::NK_ExprAttrs);
73 if (!Up)
74 return nullptr;
75
76 const SemaAttrs &SA = static_cast<const ExprAttrs &>(*Up).sema();
77 const Node *Var = SA.staticAttrs().at(AN.staticName()).value();
78 assert(Var->kind() == Node::NK_ExprVar);
79 return static_cast<const ExprVar *>(Var);
80 }
81 return nullptr;
82}

84const ExprVar *findVar(const Node &N, const ParentMapAnalysis &PMA,
85 const VariableLookupAnalysis &VLA) {
86 if (const ExprVar *InVar = findInheritVar(N, PMA, VLA))
87 return InVar;
88
89 return static_cast<const ExprVar *>(PMA.upTo(N, Node::NK_ExprVar));
90}

92Expected<Location> staticDef(URIForFile URI, const Node &N,
93 const ParentMapAnalysis &PMA,
94 const VariableLookupAnalysis &VLA) {
95 Expected<const Definition &> ExpDef = findDefinition(N, PMA, VLA);
96 if (!ExpDef)
97 return ExpDef.takeError();
98
99 if (ExpDef->isBuiltin())
100 return error("this is a builtin variable defined by nix interpreter");
101 assert(ExpDef->syntax());
102
103 return Location{
104 .uri = std::move(URI),
105 .range = toLSPRange(ExpDef->syntax()->range()),

109/// \brief Resolve definition by invoking nixpkgs provider.
110///
111/// Useful for users inspecting nixpkgs packages. For example, someone clicks
112/// "with pkgs; [ hello ]", it's better to goto nixpkgs position, instead of
113/// "with pkgs;"
114class NixpkgsDefinitionProvider {
115 AttrSetClient &NixpkgsClient;
116
117 /// \brief Parse nix-rolled location: file:line -> lsp Location
118 static Location parseLocation(std::string_view Position) {
119 // Firstly, find ":"
120 auto Pos = Position.find_first_of(':');
121 if (Pos == std::string_view::npos) {
122 return Location{
123 .uri = URIForFile::canonicalize(Position, Position),
124 .range = {{0, 0}, {0, 0}},
125 };
126 }
127 int PosL = std::stoi(std::string(Position.substr(Pos + 1)));
128 lspserver::Position P{PosL, PosL};
129 std::string_view File = Position.substr(0, Pos);
130 return Location{
131 .uri = URIForFile::canonicalize(File, File),
132 .range = {P, P},
133 };
134 }
135
136public:
137 NixpkgsDefinitionProvider(AttrSetClient &NixpkgsClient)
138 : NixpkgsClient(NixpkgsClient) {}

140 Expected<lspserver::Location> resolvePackage(std::vector<std::string> Scope,
141 std::string Name) {
142 std::binary_semaphore Ready(0);
143 Expected<PackageDescription> Desc = error("not replied");
144 auto OnReply = [&Ready, &Desc](llvm::Expected<AttrPathInfoResponse> Resp) {
145 if (Resp)
146 Desc = *Resp;
147 else
148 Desc = Resp.takeError();
149 Ready.release();
150 };
151 Scope.emplace_back(std::move(Name));
152 NixpkgsClient.attrpathInfo(Scope, std::move(OnReply));
153 Ready.acquire();
154
155 if (!Desc)
156 return Desc.takeError();
157
158 if (!Desc->Position)
159 return error("meta.position is not available for this package");
160
161 try {
162 return parseLocation(*Desc->Position);
163 } catch (std::exception &E) {
164 return error(E.what());
165 }
166 }
167};

169/// \brief Try to get "location" by invoking options worker
170class OptionsDefinitionProvider {
171 AttrSetClient &Client;
172
173public:
174 OptionsDefinitionProvider(AttrSetClient &Client) : Client(Client) {}
175 void resolveLocations(const std::vector<std::string> &Params,
176 Locations &Locs) {
177 std::binary_semaphore Ready(0);
178 Expected<OptionInfoResponse> Info = error("not replied");
179 OptionCompleteResponse Names;
180 auto OnReply = [&Ready, &Info](llvm::Expected<OptionInfoResponse> Resp) {
181 Info = std::move(Resp);
182 Ready.release();
183 };
184 // Send request.
185
186 Client.optionInfo(Params, std::move(OnReply));
187 Ready.acquire();
188
189 if (!Info) {
190 elog("getting locations: {0}", Info.takeError());
191 return;
192 }
193
194 for (const auto &Decl : Info->Declarations)
195 Locs.emplace_back(Decl);
196 }
197};

201Expected<const Definition &>
202nixd::findDefinition(const Node &N, const ParentMapAnalysis &PMA,
203 const VariableLookupAnalysis &VLA) {
204 const ExprVar *Var = findVar(N, PMA, VLA);
205 if (!Var) [[unlikely]] {
206 if (const Definition *Def = findSelfDefinition(N, PMA, VLA))
207 return *Def;
208 return error("cannot find variable on given position");
209 }
210 assert(Var->kind() == Node::NK_ExprVar);
211 LookupResult Result = VLA.query(static_cast<const ExprVar &>(*Var));
212
213 if (Result.Kind == ResultKind::Undefined)
214 return error("this varaible is undefined");
215
216 if (Result.Kind == ResultKind::NoSuchVar)
217 return error("this varaible is not used in var lookup (duplicated attr?)");
218
219 assert(Result.Def);
220
221 return *Result.Def;
222}

224void Controller::onDefinition(const TextDocumentPositionParams &Params,
225 Callback<llvm::json::Value> Reply) {
226 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
227 Pos = toNixfPosition(Params.position), this]() mutable {
228 std::string File(URI.file());
229 if (std::shared_ptr<NixTU> TU = getTU(File, Reply)) [[likely]] {
230 if (std::shared_ptr<Node> AST = getAST(*TU, Reply)) [[likely]] {
231 const VariableLookupAnalysis &VLA = *TU->variableLookup();
232 const ParentMapAnalysis &PM = *TU->parentMap();
233 const Node *N = AST->descend({Pos, Pos});
234 Locations Locs;
235 if (!N) [[unlikely]] {
236 Reply(error("cannot find AST node on given position"));
237 return;
238 }
239 using PathResult = FindAttrPathResult;
240 std::vector<std::string> Scope;
241 auto R = findAttrPath(*N, PM, Scope);
242 if (R == PathResult::OK) {
243 std::lock_guard _(OptionsLock);
244 // For each option worker, try to get it's decl position.
245 for (const auto &[_, Client] : Options) {
246 if (AttrSetClient *C = Client->client()) {
247 OptionsDefinitionProvider ODP(*C);
248 ODP.resolveLocations(Scope, Locs);
249 }
250 }
251 }
252 if (havePackageScope(*N, VLA, PM) && nixpkgsClient()) {
253 // Ask nixpkgs client what's current package documentation.
254 NixpkgsDefinitionProvider NDP(*nixpkgsClient());
255 auto [Scope, Name] = getScopeAndPrefix(*N, PM);
256 if (Expected<Location> Loc =
257 NDP.resolvePackage(std::move(Scope), std::move(Name)))
258 Locs.emplace_back(*Loc);
259 else
260 elog("cannot get nixpkgs definition for package: {0}",
261 Loc.takeError());
262 }
263
264 if (Expected<Location> StaticLoc = staticDef(URI, *N, PM, VLA))
265 Locs.emplace_back(*StaticLoc);
266 else
267 elog("cannot get static def location: {0}", StaticLoc.takeError());
268
269 if (Locs.empty()) {
270 Reply(nullptr);
271 return;
272 }
273
274 if (Locs.size() == 1) {
275 Reply(Locs.back());
276 return;
277 }
278
279 Reply(std::move(Locs));
280 return;
281 }
282 }
283 };
284 boost::asio::post(Pool, std::move(Action));
