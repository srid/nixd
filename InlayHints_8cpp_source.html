37/// Ask nixpkgs provider to compute package information, to get inlay-hints.
38class NixpkgsInlayHintsProvider {
39 AttrSetClient &NixpkgsProvider;
40 const VariableLookupAnalysis &VLA;
41 const ParentMapAnalysis &PMA;
42
43 /// Only positions contained in this range should be computed && added;
44 std::optional<nixf::PositionRange> Range;
45
46 std::vector<InlayHint> &Hints;
47
48 bool rangeOK(const nixf::PositionRange &R) {
49 if (!Range)
50 return true; // Always OK if there is no limitation.
51 return Range->contains(R);
52 }
53
54public:
55 NixpkgsInlayHintsProvider(AttrSetClient &NixpkgsProvider,
56 const VariableLookupAnalysis &VLA,
57 const ParentMapAnalysis &PMA,
58 std::optional<lspserver::Range> Range,
59 std::vector<InlayHint> &Hints)
60 : NixpkgsProvider(NixpkgsProvider), VLA(VLA), PMA(PMA), Hints(Hints) {
61 if (Range)
62 this->Range = toNixfRange(*Range);
63 }
64
65 void dfs(const Node *N) {
66 if (!N)
67 return;
68 if (N->kind() == Node::NK_ExprVar) {
69 if (havePackageScope(*N, VLA, PMA)) {
70 if (!rangeOK(N->positionRange()))
71 return;
72 // Ask nixpkgs eval to provide it's information.
73 // This is relatively slow. Maybe better query a set of packages in the
74 // future?
75 std::binary_semaphore Ready(0);
76 const std::string &Name = static_cast<const ExprVar &>(*N).id().name();
77 AttrPathInfoResponse R;
78 auto OnReply = [&Ready, &R](llvm::Expected<AttrPathInfoResponse> Resp) {
79 if (!Resp) {
80 Ready.release();
81 return;
82 }
83 R = *Resp;
84 Ready.release();
85 };
86 NixpkgsProvider.attrpathInfo({Name}, std::move(OnReply));
87 Ready.acquire();
88
89 if (R.Version) {
90 // Construct inlay hints.
91 InlayHint H{
92 .position = toLSPPosition(N->rCur()),
93 .label = ": " + *R.Version,
94 .kind = InlayHintKind::Designator,
95 .range = toLSPRange(N->range()),
96 };
97 Hints.emplace_back(std::move(H));
98 }
99 }
100 }
101 // FIXME: process other node kinds. e.g. ExprSelect.
102 for (const Node *Ch : N->children())
103 dfs(Ch);
104 }
105};

107} // namespace

109void Controller::onInlayHint(const InlayHintsParams &Params,
110 Callback<std::vector<InlayHint>> Reply) {
111 // If not enabled, exit early.
112 if (!EnableInlayHints)
113 return Reply(std::vector<InlayHint>{});
114
115 auto Action = [Reply = std::move(Reply), URI = Params.textDocument.uri,
116 Range = Params.range, this]() mutable {
117 std::string File(URI.file());
118 if (std::shared_ptr<NixTU> TU = getTU(File, Reply)) [[likely]] {
119 if (std::shared_ptr<Node> AST = getAST(*TU, Reply)) [[likely]] {
120 // Perform inlay hints computation on the range.
121 std::vector<InlayHint> Response;
122 NixpkgsInlayHintsProvider NP(*nixpkgsClient(), *TU->variableLookup(),
123 *TU->parentMap(), Range, Response);
124 NP.dfs(AST.get());
125 Reply(std::move(Response));
126 }
127 }
128 };
129 boost::asio::post(Pool, std::move(Action));
130}
