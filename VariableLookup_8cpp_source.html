43void VariableLookupAnalysis::emitEnvLivenessWarning(
44 const std::shared_ptr<EnvNode> &NewEnv) {
45 for (const auto &[Name, Def] : NewEnv->defs()) {
46 // If the definition comes from lambda arg, omit the diagnostic
47 // because there is no elegant way to "fix" this trivially & keep
48 // the lambda signature.
49 if (Def->source() == Definition::DS_LambdaArg)
50 continue;
51 if (Def->uses().empty()) {
52 Diagnostic &D = Diags.emplace_back(Diagnostic::DK_DefinitionNotUsed,
53 Def->syntax()->range());
54 D << Name;
55 D.tag(DiagnosticTag::Faded);
56 }
57 }

60void VariableLookupAnalysis::lookupVar(const ExprVar &Var,
61 const std::shared_ptr<EnvNode> &Env) {
62 const std::string &Name = Var.id().name();
63
64 bool EnclosedWith = false; // If there is a "With" enclosed this var name.
65 EnvNode *WithEnv = nullptr;
66 EnvNode *CurEnv = Env.get();
67 std::shared_ptr<Definition> Def;
68 for (; CurEnv; CurEnv = CurEnv->parent()) {
69 if (CurEnv->defs().contains(Name)) {
70 Def = CurEnv->defs().at(Name);
71 break;
72 }
73 // Find the most nested `with` expression, and set uses.
74 if (CurEnv->isWith() && !EnclosedWith) {
75 EnclosedWith = true;
76 WithEnv = CurEnv;
77 }
78 }
79
80 if (Def) {
81 Def->usedBy(Var);
82 Results.insert({&Var, LookupResult{LookupResultKind::Defined, Def}});
83
84 if (EnclosedWith) {
85 // Escaping from "with" to outer scope.
86 // https://github.com/NixOS/nix/issues/490
87
88 assert(WithEnv && "EnclosedWith -> WithEnv");
89 // Make a diagnostic.
90 Diagnostic &D =
91 Diags.emplace_back(Diagnostic::DK_EscapingWith, Var.range());
92 if (Def->syntax()) {
93 D.note(Note::NK_VarBindToThis, Def->syntax()->range());
94 }
95 const auto &KwWith =
96 static_cast<const nixf::ExprWith *>(WithEnv->syntax())->kwWith();
97 D.note(Note::NK_EscapingWith, KwWith.range());
98 }
99 return;
100 }
101 if (EnclosedWith) {
102 Def = WithDefs.at(WithEnv->syntax());
103 Def->usedBy(Var);
104 Results.insert({&Var, LookupResult{LookupResultKind::FromWith, Def}});
105 return;
106 }
107
108 // Otherwise, this variable is undefined.
109 Results.insert({&Var, LookupResult{LookupResultKind::Undefined, nullptr}});
110 Diagnostic &Diag =
111 Diags.emplace_back(Diagnostic::DK_UndefinedVariable, Var.range());
112 Diag << Var.id().name();
113}

115void VariableLookupAnalysis::dfs(const ExprLambda &Lambda,
116 const std::shared_ptr<EnvNode> &Env) {
117 // Early exit for in-complete lambda.
118 if (!Lambda.body())
119 return;
120
121 // Create a new EnvNode, as lambdas may have formal & arg.
122 DefBuilder DBuilder;
123 assert(Lambda.arg());
124 const LambdaArg &Arg = *Lambda.arg();
125
126 // foo: body
127 // ^~~<------- add function argument.
128 if (Arg.id()) {
129 // Function arg cannot duplicate to it's formal.
130 // If it this unluckily happens, we would like to skip this definition.
131 if (!Arg.formals() || !Arg.formals()->dedup().contains(Arg.id()->name()))
132 ToDef.insert_or_assign(Arg.id(), DBuilder.add(Arg.id()->name(), Arg.id(),
133 Definition::DS_LambdaArg));
134 }
135
136 // { foo, bar, ... } : body
137 /// ^~~~~~~~~<-------------- add function formals.
138 if (Arg.formals())
139 for (const auto &[Name, Formal] : Arg.formals()->dedup())
140 ToDef.insert_or_assign(
141 Formal->id(),
142 DBuilder.add(Name, Formal->id(), Definition::DS_LambdaFormal));
143
144 auto NewEnv = std::make_shared<EnvNode>(Env, DBuilder.finish(), &Lambda);
145
146 if (Arg.formals()) {
147 for (const auto &Formal : Arg.formals()->members()) {
148 if (const Expr *Def = Formal->defaultExpr()) {
149 dfs(*Def, NewEnv);
150 }
151 }
152 }
153
154 dfs(*Lambda.body(), NewEnv);
155
156 emitEnvLivenessWarning(NewEnv);
157}

170std::shared_ptr<EnvNode> VariableLookupAnalysis::dfsAttrs(
171 const SemaAttrs &SA, const std::shared_ptr<EnvNode> &Env,
172 const Node *Syntax, Definition::DefinitionSource Source) {
173 if (SA.isRecursive()) {
174 // rec { }, or let ... in ...
175 DefBuilder DB;
176 // For each static names, create a name binding.
177 for (const auto &[Name, Attr] : SA.staticAttrs())
178 ToDef.insert_or_assign(&Attr.key(), DB.add(Name, &Attr.key(), Source));
179
180 auto NewEnv = std::make_shared<EnvNode>(Env, DB.finish(), Syntax);
181
182 for (const auto &[_, Attr] : SA.staticAttrs()) {
183 if (!Attr.value())
184 continue;
185 dfs(*Attr.value(), NewEnv);
186 }
187
188 dfsDynamicAttrs(SA.dynamicAttrs(), NewEnv);
189 return NewEnv;
190 }
191
192 // Non-recursive. Dispatch nested node with old Env
193 for (const auto &[_, Attr] : SA.staticAttrs()) {
194 if (!Attr.value())
195 continue;
196 dfs(*Attr.value(), Env);
197 }
198
199 dfsDynamicAttrs(SA.dynamicAttrs(), Env);
200 return Env;
201};

203void VariableLookupAnalysis::dfs(const ExprAttrs &Attrs,
204 const std::shared_ptr<EnvNode> &Env) {
205 const SemaAttrs &SA = Attrs.sema();
206 std::shared_ptr<EnvNode> NewEnv =
207 dfsAttrs(SA, Env, &Attrs, Definition::DS_Rec);
208 if (NewEnv != Env) {
209 assert(Attrs.isRecursive() &&
210 "NewEnv must be created for recursive attrset");
211 if (!NewEnv->isLive()) {
212 Diagnostic &D = Diags.emplace_back(Diagnostic::DK_ExtraRecursive,
213 Attrs.rec()->range());
214 D.fix("remove `rec` keyword")
215 .edit(TextEdit::mkRemoval(Attrs.rec()->range()));
216 D.tag(DiagnosticTag::Faded);
217 }
218 }

243void VariableLookupAnalysis::dfs(const ExprWith &With,
244 const std::shared_ptr<EnvNode> &Env) {
245 auto NewEnv = std::make_shared<EnvNode>(Env, EnvNode::DefMap{}, &With);
246 if (!WithDefs.contains(&With)) {
247 auto NewDef =
248 std::make_shared<Definition>(&With.kwWith(), Definition::DS_With);
249 ToDef.insert_or_assign(&With.kwWith(), NewDef);
250 WithDefs.insert_or_assign(&With, NewDef);
251 }
252
253 if (With.with())
254 dfs(*With.with(), Env);
255
256 if (With.expr())
257 dfs(*With.expr(), NewEnv);
258
259 if (WithDefs.at(&With)->uses().empty()) {
260 Diagnostic &D =
261 Diags.emplace_back(Diagnostic::DK_ExtraWith, With.kwWith().range());
262 Fix &F = D.fix("remove `with` expression")
263 .edit(TextEdit::mkRemoval(With.kwWith().range()));
264 if (With.tokSemi())
265 F.edit(TextEdit::mkRemoval(With.tokSemi()->range()));
266 if (With.with())
267 F.edit(TextEdit::mkRemoval(With.with()->range()));
268 }

271void VariableLookupAnalysis::dfs(const Node &Root,
272 const std::shared_ptr<EnvNode> &Env) {
273 Envs.insert({&Root, Env});
274 switch (Root.kind()) {
275 case Node::NK_ExprVar: {
276 const auto &Var = static_cast<const ExprVar &>(Root);
277 lookupVar(Var, Env);
278 break;
279 }
280 case Node::NK_ExprLambda: {
281 const auto &Lambda = static_cast<const ExprLambda &>(Root);
282 dfs(Lambda, Env);
283 break;
284 }
285 case Node::NK_ExprAttrs: {
286 const auto &Attrs = static_cast<const ExprAttrs &>(Root);
287 dfs(Attrs, Env);
288 break;
289 }
290 case Node::NK_ExprLet: {
291 const auto &Let = static_cast<const ExprLet &>(Root);
292 dfs(Let, Env);
293 break;
294 }
295 case Node::NK_ExprWith: {
296 const auto &With = static_cast<const ExprWith &>(Root);
297 dfs(With, Env);
298 break;
299 }
300 default:
301 trivialDispatch(Root, Env);
302 }
