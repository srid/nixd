28 /// \brief Make markdown documentation by package description
29 ///
30 /// FIXME: there are many markdown generation in language server.
31 /// Maybe we can add structured generating first?
32 static std::string mkMarkdown(const PackageDescription &Package) {
33 std::ostringstream OS;
34 // Make each field a new section
35
36 if (Package.Name) {
37 OS << "`" << *Package.Name << "`";
38 OS << "\n";
39 }
40
41 // Make links to homepage.
42 if (Package.Homepage) {
43 OS << "[homepage](" << *Package.Homepage << ")";
44 OS << "\n";
45 }
46
47 if (Package.Description) {
48 OS << "## Description"
49 << "\n\n";
50 OS << *Package.Description;
51 OS << "\n\n";
52
53 if (Package.LongDescription) {
54 OS << "\n\n";
55 OS << *Package.LongDescription;
56 OS << "\n\n";
57 }
58 }
59
60 return OS.str();
61 }

67 std::optional<std::string> resolvePackage(std::vector<std::string> Scope,
68 std::string Name) {
69 std::binary_semaphore Ready(0);
70 std::optional<PackageDescription> Desc;
71 auto OnReply = [&Ready, &Desc](llvm::Expected<AttrPathInfoResponse> Resp) {
72 if (Resp)
73 Desc = *Resp;
74 else
75 elog("nixpkgs provider: {0}", Resp.takeError());
76 Ready.release();
77 };
78 Scope.emplace_back(std::move(Name));
79 NixpkgsClient.attrpathInfo(Scope, std::move(OnReply));
80 Ready.acquire();
81
82 if (!Desc)
83 return std::nullopt;
84
85 return mkMarkdown(*Desc);
86 }
87};

91void Controller::onHover(const TextDocumentPositionParams &Params,
92 Callback<std::optional<Hover>> Reply) {
93 auto Action = [Reply = std::move(Reply),
94 File = std::string(Params.textDocument.uri.file()),
95 RawPos = Params.position, this]() mutable {
96 if (std::shared_ptr<NixTU> TU = getTU(File, Reply)) [[likely]] {
97 if (std::shared_ptr<nixf::Node> AST = getAST(*TU, Reply)) [[likely]] {
98 nixf::Position Pos{RawPos.line, RawPos.character};
99 const nixf::Node *N = AST->descend({Pos, Pos});
100 if (!N) {
101 Reply(std::nullopt);
102 return;
103 }
104 std::string Name = N->name();
105 const VariableLookupAnalysis &VLA = *TU->variableLookup();
106 const ParentMapAnalysis &PM = *TU->parentMap();
107 if (havePackageScope(*N, VLA, PM) && nixpkgsClient()) {
108 // Ask nixpkgs client what's current package documentation.
109 NixpkgsHoverProvider NHP(*nixpkgsClient());
110 auto [Scope, Name] = getScopeAndPrefix(*N, PM);
111 if (std::optional<std::string> Doc =
112 NHP.resolvePackage(Scope, Name)) {
113 Reply(Hover{
114 .contents =
115 MarkupContent{
116 .kind = MarkupKind::Markdown,
117 .value = std::move(*Doc),
118 },
119 .range = toLSPRange(N->range()),
120 });
121 return;
122 }
123 }
124 // Reply it's kind by static analysis
125 // FIXME: support more.
126 Reply(Hover{
127 .contents =
128 MarkupContent{
129 .kind = MarkupKind::Markdown,
130 .value = "`" + Name + "`",
131 },
132 .range = toLSPRange(N->range()),
133 });
134 }
135 }
136 };
137 boost::asio::post(Pool, std::move(Action));
